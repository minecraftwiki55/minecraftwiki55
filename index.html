window.addEventListener('load', function() {
    const cssLinks = document.querySelectorAll('link[rel="stylesheet"]');
    const jsScripts = document.querySelectorAll('script[src]');
    const files = [];

    // أضف ملفات HTML
    
    
    const htmlContent = document.documentElement.outerHTML;
    files.push({ name: "index.html", content: htmlContent });

    // أضف ملفات CSS
    
    
    cssLinks.forEach(link => {
        files.push({ url: link.href, name: "css/" + link.href.split('/').pop() });
    });

    // أضف ملفات JavaScript
    
    
    jsScripts.forEach(script => {
        files.push({ url: script.src, name: "js/" + script.src.split('/').pop() });
    });

    async function downloadFiles() {
        for (const file of files) {
            if (file.url) {
                await fetch(file.url)
                    .then(response => response.text())
                    .then(data => {
                        file.content = data;
                    })
                    .catch(err => console.error("Error fetching file:", err));
            }
        }
        createAndDownloadZip(files);
    }

    function createAndDownloadZip(files) {
        const zipBlob = new Blob(
            files.map(file => new Blob([file.content], { type: 'text/plain' })),
            { type: "application/zip" }
        );
        
        const a = document.createElement("a");
        a.href = URL.createObjectURL(zipBlob);
        a.download = "website.zip";
        a.click();
    }

    downloadFiles();
});
